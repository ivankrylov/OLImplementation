  37 inline void oopDesc::push_contents(PSPromotionManager* pm) {
  38   Klass* k = klass();
  39   if (!k->oop_is_typeArray()) {
  40     // It might contain oops beyond the header, so take the virtual call.
  41     k->oop_push_contents(pm, this);
  42   }
  43   // Else skip it.  The TypeArrayKlass in the header never needs scavenging.
  44 }
